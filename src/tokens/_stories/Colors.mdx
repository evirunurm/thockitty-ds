import { Meta, Unstyled } from '@storybook/addon-docs/blocks'
import { tokens } from '../ts/colors'
import { ColorSwatch } from './_components/color-swatch/ColorSwatch'
import { Section } from './_components/section/Section'
import { DocPage } from './_components/doc-page/DocPage'

export const groups = tokens.reduce((acc, token) => {
	const group = token.path.split('.')[2] ?? 'other'
	;(acc[group] ??= []).push(token)
	return acc
}, {})

<Meta title="Tokens/Colors" />

<Unstyled>
<DocPage>
	{Object.entries(groups).map(([group, groupTokens]) => (
		<Section key={group} title={group} bordered padded>
			<div
				style={{
					display: 'grid',
					gridTemplateColumns: 'repeat(auto-fill, minmax(200px, 1fr))',
					gap: 'var(--foundation-space-006)',
				}}
			>
				{groupTokens.map((token) => (
					<ColorSwatch key={token.name} color={token} />
				))}
			</div>
		</Section>
	))}
</DocPage>
</Unstyled>
